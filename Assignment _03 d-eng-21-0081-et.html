import cv2
import numpy as np
from sklearn.linear_model import RANSACRegressor
import matplotlib.pyplot as plt

img = cv2.imread('/examples/building_crop.jpg')
gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
edges = cv2.Canny(gray, 350, 390)
indices = np.where(edges != [0])
x = indices[1]
y = -indices[0] 

model_ransac = RANSACRegressor(base_estimator=None, min_samples=2, residual_threshold=3.0, max_trials=100, random_state=None)
model_ransac.fit(x.reshape(-1, 1), y)
slope = model_ransac.estimator_.coef_[0]
angle = np.arctan(slope) * 180 / np.pi 
print("The angle when we use this method is : " ,angle)
line_y = np.linspace(np.min(y), np.max(y), len(x))
line_x = np.round(x[0] + (line_y - y[0]) / slope).astype(int)

plt.figure(figsize=(10,5))
plt.scatter(x, y, s=1, c='black')
plt.plot(line_x, line_y, linewidth=2, c='red')
plt.title('Edge Points and Estimated Roof Slope (using RANSAC)')
plt.xlabel('x')
plt.ylabel('y')
plt.gca() 
plt.show()

# The provided angle is almost accurate and there is a difference between LS and RANSAC that can be clearly seen by the angle calculation.

## Explain, why your proposed approach is performing better than the least-squares-fit and total
## least-squares-fit.
#RANSAC method was  designed to handle outliers in the data,
#  by iteratively fitting a model to a random subset of the data and identifying
#  the remaining points as "outliers". The inliers are then used to calculate a robust estimate of the model parameters,
#  which is less affected by the presence of outliers. This process is repeated a fixed number of times,
#  and the best model found during the iterations is returned as the final result. As a result, 
#  RANSAC is expected to provide a more accurate estimate of the line slope in the presence of outliers
#  compared to the least-squares and total least-squares methods.
